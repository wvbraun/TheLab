# Define a procedure, greatest,
# that takes as input a list
# of positive numbers, and
# returns the greatest number
# in that list. If the input
# list is empty, the output
# should be 0.

def greatest(n):
    if n == []:
        return 0 
    else:
        return max(n)

#or 
def greatest(n):
    big = 0
    for e in n:
        if e > big:
            big = e 
    return big 




print greatest([4,23,1])
#>>> 23
print greatest([])
#>>> 0
