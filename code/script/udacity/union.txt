# Define a procedure, union,
# that takes as inputs two lists.
# It should modify the first input
# list to be the set union of the two
# lists. You may assume the first list
# is a set, that is, it contains no 
# repeated elements.


def union(x,y):
    
    for e in x:
        for e in y:
            if e not in x:
                x.append(e)
    return x
