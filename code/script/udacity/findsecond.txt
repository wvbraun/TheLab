# Define a procedure, find_second, that takes
# two strings as its inputs: a search string
# and a target string. It should return a
# number that is the position of the second
# occurrence of the target string in the
# search string.



def find_second(name, string):
       first= name.find(string)
       second = first +1
      
       return name.find(string, second)

# or you could do this (cleaner)

def find_second(name, string):
	first = name.find(string)
	second = name.find(string, first+1)
	return second

#or (ehh)

def find_search(name, string):
	return name.find(string, name.find(string)+1)




danton = "De l'audace, encore de l'audace, toujours de l'audace"
